A scalar currency variable is defined as

\begin{minted}[fontsize=\footnotesize]{xml}
      <Currency>
        <Name>PayCcy</Name>
        <Value>USD</Value>
      </Currency>
\end{minted}

and an array of currencies as

\begin{minted}[fontsize=\footnotesize]{xml}
      <Currency>
        <Name>BasketCurrencies</Name>
        <Values>
          <Value>USD</Value>
          <Value>GBP</Value>
          <Value>JPY</Value>
        </Value>
      </Currency>
\end{minted}
